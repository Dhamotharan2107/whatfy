# =======================================
def cleanup_lock_files(user_data_dir):
    """
    Remove Chrome lock files that might prevent the session from starting.
    """
    lock_files = [
        'SingletonLock',
        'SingletonSocket',
        'SingletonCookie',
        'lockfile'
    ]

    print(f"Cleaning up lock files in: {user_data_dir}")

    for lock_file in lock_files:
        lock_path = os.path.join(user_data_dir, lock_file)
        if os.path.exists(lock_path):
            try:
                if os.path.isdir(lock_path):
                    shutil.rmtree(lock_path)
                else:
                    os.remove(lock_path)
                print(f"  Removed: {lock_file}")
            except Exception as e:
                print(f"  Could not remove {lock_file}: {e}")

def open_whatsapp_group(group_name):
    """
    Opens the specified WhatsApp group by searching for it.
    """
    try:
        # Wait a bit for WhatsApp to fully load
        t.sleep(2)

        # Search for the group - try multiple XPath options
        search_xpaths = [
            "//div[@contenteditable='true'][@data-tab='3']",  # Search box
            "//div[@contenteditable='true' and @role='textbox']",
            "//div[@title='Search input textbox']",
            "//p[@class='selectable-text copyable-text'][@contenteditable='true']",
            "//div[@contenteditable='true'][@data-lexical-editor='true']"
        ]

        input_bar = None
        for xpath in search_xpaths:
            try:
                input_bar = WebDriverWait(driver, 10).until(
                    EC.element_to_be_clickable((By.XPATH, xpath))
                )
                print(f"Search bar found with XPath: {xpath}")
                break
            except:
                continue

        if input_bar is None:
            raise Exception("Could not find search bar")

        input_bar.click()
        t.sleep(1)
        input_bar.clear()
        input_bar.send_keys(group_name)
        t.sleep(2)
        input_bar.send_keys(Keys.RETURN)
        t.sleep(3)

        # Select the group
        collectiongroup = WebDriverWait(driver, 30).until(
            EC.element_to_be_clickable((By.XPATH, f"//span[@title='{group_name}']")),
            message=f'Group "{group_name}" not found'
        )
        collectiongroup.click()
        print(f"Group '{group_name}' selected successfully")
        t.sleep(3)
    except TimeoutException as e:
        logging.error(f"Timeout during WhatsApp initialization: {e}")
        print(f"Error: {e}")
        raise
    except Exception as e:
        logging.error(f"Unexpected error during WhatsApp initialization: {e}")
        print(f"Unexpected error: {e}")
        raise

def generate_table_image(df, image_name, total_label="TOTAL"):
    """
    Generates a table image from a DataFrame and saves it as an image file.
    FIXED: Ensures minimum dimensions to prevent sticker conversion.
    """
    if 'Amount' in df.columns:
        df['Amount'] = pd.to_numeric(df['Amount'], errors='coerce').fillna(0)
        df = df.sort_values(by='Amount', ascending=False).reset_index(drop=True)
        print(f"DataFrame sorted by 'Amount':\n{df}")
    else:
        print("'Amount' column not found in DataFrame. Skipping sort.")

    cell_padding = 20  # Increased padding
    cell_width = 250  # Increased width
    row_height = 50  # Increased height
    header_color = (255, 165, 0)
    row_colors = [(255, 255, 255), (224, 238, 238)]
    text_color = (0, 0, 0)

    rows = len(df)
    cols = len(df.columns)

    table_width = (cols + 1) * cell_width
    table_height = (rows + 2) * row_height

    # *** IMPORTANT: Ensure minimum dimensions (WhatsApp treats small/square images as stickers) ***
    min_width = 800   # Increased from 600
    min_height = 600  # Increased from 400

    final_width = max(table_width + 2 * cell_padding, min_width)
    final_height = max(table_height + 2 * cell_padding, min_height)

    img = Image.new('RGB', (final_width, final_height), color=(255, 255, 255))
    draw = ImageDraw.Draw(img)

    try:
        font_path = "C:/Windows/Fonts/arial.ttf"
        font = ImageFont.truetype(font_path, size=20)  # Slightly larger font
    except OSError:
        print("Font not found; using default font.")
        font = ImageFont.load_default()

    # Draw table headers
    for col_idx in range(cols):
        x = col_idx * cell_width + cell_padding
        y = cell_padding
        draw.rectangle([x, y, x + cell_width, y + row_height], fill=header_color)
        header_text = str(df.columns[col_idx]).upper()
        draw.text((x + 10, y + 10), header_text, font=font, fill=text_color)

    # Draw table rows
    for row_idx in range(rows):
        for col_idx in range(cols):
            x = col_idx * cell_width + cell_padding
            y = (row_idx + 1) * row_height + cell_padding
            color = row_colors[row_idx % len(row_colors)]
            draw.rectangle([x, y, x + cell_width, y + row_height], fill=color)
            cell_value = str(df.iloc[row_idx, col_idx])
            draw.text((x + 10, y + 10), cell_value, font=font, fill=text_color)

    # Draw the TOTAL row
    x = cell_padding
    y = (rows + 1) * row_height + cell_padding
    draw.rectangle([x, y, x + cell_width, y + row_height], fill=row_colors[1])
    draw.text((x + 10, y + 10), total_label, font=font, fill=text_color)

    total_amount = df['Amount'].sum()
    draw.text((x + cell_width + 10, y + 10), f"{total_amount:.2f}", font=font, fill=text_color)

    # Save as JPEG with high quality (PNG can sometimes be treated as sticker)
    img.save(image_name, 'JPEG', quality=95, dpi=(300, 300))
    print(f"Table image saved as '{image_name}' ({final_width}x{final_height}px).")
    return total_amount

#   SIMPLE: DIRECT FILE UPLOAD METHOD
# =======================================
def send_image_via_direct_upload(image_path, max_retries=3):
    """
    Simplified method: Skip attach menu, go directly to file input.
    Add caption to prevent sticker conversion.
    """
    for attempt in range(max_retries):
        try:
            print(f"Attempt {attempt + 1}/{max_retries}: Using direct upload method...")

            # Look for file input directly (sometimes it's already present)
            file_input_xpaths = [
                "//input[@accept='image/*,video/mp4,video/3gpp,video/quicktime']",
                "//input[@type='file' and contains(@accept, 'image')]",
                "//input[@type='file']"
            ]

            # First try to find existing file input
            image_input = None
            for xpath in file_input_xpaths:
                try:
                    image_input = driver.find_element(By.XPATH, xpath)
                    if image_input.is_displayed():
                        print(f"✓ Found visible file input with XPath: {xpath}")
                        break
                    else:
                        # Make it visible if it's hidden
                        driver.execute_script("arguments[0].style.display = 'block'; arguments[0].style.visibility = 'visible';", image_input)
                        print(f"✓ Made file input visible with XPath: {xpath}")
                        break
                except:
                    continue

            # If no file input found, click attach first
            if image_input is None:
                print("No direct file input found, clicking attach button first...")
                attach_button = WebDriverWait(driver, 5).until(
                    EC.element_to_be_clickable((By.XPATH, "//div[@title='Attach'] | //span[@data-icon='plus']/.. | //button[@aria-label='Attach']"))
                )
                driver.execute_script("arguments[0].click();", attach_button)
                t.sleep(2)

                # Now look for file input again
                for xpath in file_input_xpaths:
                    try:
                        image_input = WebDriverWait(driver, 5).until(
                            EC.presence_of_element_located((By.XPATH, xpath))
                        )
                        print(f"✓ File input found after attach click: {xpath}")
                        break
                    except:
                        continue

            if image_input is None:
                raise Exception("Could not find file input")

            # Send the file
            absolute_path = os.path.abspath(image_path)
            if not os.path.exists(absolute_path):
                raise Exception(f"Image file does not exist: {absolute_path}")

            image_input.send_keys(absolute_path)
            print(f"✓ Image path sent: {absolute_path}")
            t.sleep(3)

            # Add caption to prevent sticker conversion
            try:
                caption_input = WebDriverWait(driver, 5).until(
                    EC.presence_of_element_located((By.XPATH, "//div[@contenteditable='true'][@data-tab='10'] | //div[@role='textbox' and @contenteditable='true']"))
                )
                caption_input.send_keys("📊 Report")
                print("✓ Caption added to prevent sticker conversion")
            except:
                print("⚠ Could not add caption - may send as sticker")

            t.sleep(2)

            # Click send
            send_button = WebDriverWait(driver, 10).until(
                EC.element_to_be_clickable((By.XPATH, "//span[@data-icon='send'] | //button[@aria-label='Send']"))
            )
            driver.execute_script("arguments[0].click();", send_button)
            print(f"✓ Image sent via direct upload method!")
            t.sleep(3)
            return True

        except Exception as e:
            print(f"✗ Direct upload method failed on attempt {attempt + 1}: {e}")
            if attempt < max_retries - 1:
                t.sleep(3)
            else:
                return False

    return False

def send_image_via_drag_drop(image_path, max_retries=3):
    """
    Alternative method: Send image using drag and drop to message input area.
    This method can sometimes avoid sticker conversion.
    """
    for attempt in range(max_retries):
        try:
            print(f"Attempt {attempt + 1}/{max_retries}: Using drag and drop method...")

            # Find the message input area
            message_input_xpaths = [
                "//div[@contenteditable='true'][@data-tab='10']",
                "//div[@contenteditable='true' and @role='textbox']",
                "//footer//div[@contenteditable='true']",
                "//div[@contenteditable='true'][@data-lexical-editor='true']"
            ]

            message_input = None
            for xpath in message_input_xpaths:
                try:
                    message_input = WebDriverWait(driver, 10).until(
                        EC.presence_of_element_located((By.XPATH, xpath))
                    )
                    print(f"✓ Message input found with XPath: {xpath}")
                    break
                except:
                    continue

            if message_input is None:
                raise Exception("Could not find message input area")

            # Create a hidden file input element and trigger it
            absolute_path = os.path.abspath(image_path)
            
            # Ensure the file exists
            if not os.path.exists(absolute_path):
                raise Exception(f"Image file does not exist: {absolute_path}")

            # Use JavaScript to create a file input and simulate file selection
            js_script = """
            var input = document.createElement('input');
            input.type = 'file';
            input.accept = 'image/*';
            input.style.display = 'none';
            document.body.appendChild(input);
            
            var event = new Event('change', { bubbles: true });
            var file = new File([''], arguments[0].split('/').pop(), { type: 'image/jpeg' });
            
            // Create a DataTransfer object and add the file
            var dt = new DataTransfer();
            dt.items.add(file);
            input.files = dt.files;
            
            // Trigger the change event
            input.dispatchEvent(event);
            
            return input;
            """
            
            # Execute the JavaScript
            file_input = driver.execute_script(js_script, absolute_path)
            
            # Send the file path to the created input
            file_input.send_keys(absolute_path)
            print(f"✓ File path sent via JavaScript: {absolute_path}")
            
            t.sleep(3)
            
            # Look for and click send button
            send_xpaths = [
                "//span[@data-icon='send']",
                "//button[@aria-label='Send']",
                "//*[@data-icon='send']",
                "//div[@aria-label='Send']"
            ]
            
            send_button = None
            for xpath in send_xpaths:
                try:
                    send_button = WebDriverWait(driver, 5).until(
                        EC.element_to_be_clickable((By.XPATH, xpath))
                    )
                    driver.execute_script("arguments[0].click();", send_button)
                    print(f"✓ Image sent via drag and drop method!")
                    t.sleep(3)
                    return True
                except:
                    continue
                    
            if send_button is None:
                print("⚠ Send button not found after drag and drop")
                return False

        except Exception as e:
            print(f"✗ Drag and drop method failed on attempt {attempt + 1}: {e}")
            if attempt < max_retries - 1:
                t.sleep(3)
            else:
                return False

    return False

# =======================================
#   FUNCTION: SEND WHATSAPP TEXT MESSAGE
# =======================================
def send_whatsapp_msg(text, max_retries=3):
    """
    Send WhatsApp text message with retry logic and updated XPath selectors.
    """
    for attempt in range(max_retries):
        try:
            # Multiple XPath options for message input box
            message_xpaths = [
                "//div[@contenteditable='true'][@data-tab='10']",
                "//div[@contenteditable='true' and @role='textbox']",
                "//p[@class='selectable-text copyable-text'][@contenteditable='true']",
                "//div[@title='Type a message']",
                "//footer//div[@contenteditable='true']",
                "//div[@contenteditable='true'][@data-lexical-editor='true']"
            ]

            chat_input = None
            for xpath in message_xpaths:
                try:
                    chat_input = WebDriverWait(driver, 10).until(
                        EC.element_to_be_clickable((By.XPATH, xpath))
                    )
                    break
                except:
                    continue

            if chat_input is None:
                raise Exception("Could not find message input box")

            chat_input.click()
            t.sleep(1)
            chat_input.send_keys(text)
            chat_input.send_keys(Keys.RETURN)
            print("✓ Text message sent successfully!")
            t.sleep(2)
            return True

        except TimeoutException:
            print(f"✗ Timeout waiting for message box (attempt {attempt + 1}/{max_retries})")
            if attempt < max_retries - 1:
                t.sleep(3)
            else:
                return False

        except InvalidElementStateException:
            print(f"✗ Element not in a state to send messages (attempt {attempt + 1}/{max_retries})")
            if attempt < max_retries - 1:
                t.sleep(3)
            else:
                return False

        except Exception as e:
            print(f"✗ Unexpected error sending text (attempt {attempt + 1}/{max_retries}): {e}")
            if attempt < max_retries - 1:
                t.sleep(3)
            else:
                return False

    return False

#    FUNCTION: CLEAR IMAGE + DATAFRAME
# =======================================
def clear_data(image_paths, df=None):
    """
    Delete image files. If df is provided, clear it.
    """
    if isinstance(image_paths, str):
        image_paths = [image_paths]

    for image_path in image_paths:
        if os.path.isfile(image_path):
            try:
                os.remove(image_path)
                print(f"✓ Deleted image: {image_path}")
            except Exception as e:
                print(f"✗ Error deleting {image_path}: {e}")
        else:
            print(f"⚠ Warning: {image_path} not found.")

    if df is not None:
        df.drop(df.index, inplace=True)
        print("✓ DataFrame cleared!")
        return df
